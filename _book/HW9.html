# ------------------------------------------------------------
# CV-MSE LADDER
# ------------------------------------------------------------

k <- 5
folds <- sample(rep(1:k, length.out = nrow(nhanes_new_data)))

cv_mse <- function(type) {
  mse <- numeric(k)
  for (i in 1:k) {
    train <- nhanes_new_data[folds != i, ]
    test  <- nhanes_new_data[folds == i, ]
    
    if (type == "shared") {
      f <- ols(Testosterone ~ rcs(BMI, 5), data = train)
      p <- predict(f, test)
      
    } else if (type == "offset") {
      f <- ols(Testosterone ~ rcs(BMI, 5) + HealthGen, data = train)
      p <- predict(f, test)
      
    } else if (type == "interaction") {
      f <- ols(Testosterone ~ rcs(BMI, 5) * HealthGen, data = train)
      p <- predict(f, test)
      
    } else {  # fully separate
      p <- rep(NA_real_, nrow(test))
      for (g in levels(train$HealthGen)) {
        tr  <- train[train$HealthGen == g, ]
        idx <- which(test$HealthGen == g)
        if (nrow(tr) >= 20 && length(idx) > 0) {
          p[idx] <- predict(ols(Testosterone ~ rcs(BMI, 5), data = tr), test[idx, ])
        } else {
          p[idx] <- mean(tr$Testosterone, na.rm = TRUE)
        }
      }
    }
    mse[i] <- mean((test$Testosterone - p)^2, na.rm = TRUE)
  }
  mean(mse)
}

results <- data.frame(
  Model  = c("1. Fully shared",

             "4. Fully separate"),
  CV_MSE = c(cv_mse("shared"),
